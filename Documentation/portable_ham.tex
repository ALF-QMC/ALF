\subsection{The interactions: \texttt{potentials.txt}}

\begin{equation}
\hat{H}_I =  \sum_{\ve{i}} \sum_{k=1}^N  U_k \left\{ \sum_{s=1}^{N_{\mathrm{fl}}} \sum_{\sigma=1}^{N_{\mathrm{col}}} \left[ \sum_{n=1}^{M_{k,s}} \left(
 \hat{c}^{\dagger}_{x^{(n)},s,\sigma} V^{(k,s,n)} \hat{c}^{\phantom{\dagger}}_{y^{(n)},s,\sigma} + \hc
\right) + \alpha_{k,s} \right] \right\}^2
\end{equation}

\begin{aligned}
x^{(n)}&=(\ve{i}+n^{(k,s,n)}_1 \bm{a}_1 + n^{(k,s,n)}_2 \bm{a}_2,\beta^{(k,s,n)}_{1}) \\
